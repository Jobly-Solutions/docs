## Atributos / Props

<Tabs>
  <Tab title="HTML" >
    ```html
    <script type="module">
  import Form from 'https://cdn.jsdelivr.net/npm/@bravilo/embeds/dist/form/index.js';

Form.initStandard({
  formId: 'TU_FORMULARIO_ID',

  onEnd: function (values) {
    console.log('Formulario enviado con valores:', values);
  },
})


</script>
    ```
  </Tab>

<ParamField path="formId" type="string" required>
  ID del Formulario
</ParamField>


<ParamField path="onEnd" type="function">
  Función de callback que se llama cuando el formulario es enviado. La función recibe los valores del formulario como un objeto.
```js
const form = await Form.initBubble({
  agentId: 'TU_FORMULARIO_ID',
  onEnd: function (values) {
    console.log('Formulario enviado con valores:', values);
  },
})
`````
</ParamField>
